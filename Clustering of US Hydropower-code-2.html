import pandas as pd

from sklearn.preprocessing import StandardScaler

import matplotlib.pyplot as plt
import matplotlib.cm as cm
import seaborn as sns
import warnings

def scale_df(df, scaler):
    return pd.DataFrame(scaler.fit_transform(df),columns=df.columns)

def pca_transform(df, npc):
    pcdf = PCA(npc).fit(df).transform(df)
    return pd.DataFrame(pcdf, columns = ["PC{}".format(x+1) for x in range(npc)])

def screeplot(df):
    pca = PCA().fit(df)
    fig1, ax1 = plt.subplots()
    ax1.title.set_text("Principal Components Compared to Variance Explained")
    ax1.plot(range(pca.n_components_), pca.explained_variance_)

def sweep_clustering(range_n_clusters, X, metric='Silhouette'):
    
    m = {}
    
    # choose best k
    for k in range_n_clusters:
        clusterer = KMeans(n_clusters=k, random_state=10)
        cluster_labels = clusterer.fit_predict(X)
        
        if metric == 'Silhouette':
            # uses the 'Silhouette' Metric
            sil_avg = silhouette_score(X, cluster_labels)
            m[sil_avg] = k
            
        elif metric == 'Pseudo_f':
            # uses the 'Pseudo-F' Metric
            ps_f = calinski_harabasz_score(X, cluster_labels)
            m[ps_f] = k
            

    # return cluster_labels of best k
    k = m[max(m.keys())]
    
    #plot metrics
    fig1, ax1 = plt.subplots()
    ax1.title.set_text("{} Score for k Clusters".format(metric))
    ax1.plot(list(m.values()), list(m.keys())) 
    
    clusterer = KMeans(n_clusters=k, random_state=10)
    return clusterer.fit_predict(X) 

def cluster_dists(df):
    """prints up to 10 clusters.
    Removes 'noise' clusters from plot"""
    
    num_cols = [x for x in df.columns if x != "cluster_label"]
    f, axes = plt.subplots(1, len(num_cols), figsize=(20,5))
    for i in range(len(axes)):
        noise_clusters = 0
        for x in df.groupby('cluster_label'):
            if x[0] >= len(sns.color_palette()):
                continue
            if not np.var(x[1][num_cols[i]]):
                noise_clusters += 1
                continue
            clust_color = sns.color_palette()[x[0]]
            ax = sns.kdeplot(x[1][num_cols[i]], shade=True, color=clust_color, ax=axes[i])
            ax.get_legend().remove()
            ax.title.set_text("{} with {} Clusters".format(num_cols[i], x[0]-noise_clusters+1))
            
def plot_dists(df, name):
    cols = [x for x in df.columns if np.var(df[x])]
    f, axes = plt.subplots(1, len(cols), figsize=(20,5))
    
    for i in range(len(axes)):
        try:
            ax = sns.kdeplot(df[cols[i]], shade=True, ax=axes[i])
            ax.get_legend().remove()
            ax.title.set_text("{}:{}".format(name, cols[i]))
        except Warning:
            continue

        

def describe_clusters(df):
#     df.groupby('cluster_label').median().T.reindex()
    desc = df.groupby('cluster_label').describe().reindex()
    desc = desc[[x for x in desc.columns if ('%' not in x[1])]]
    # have an inverse transform PCA?
    return desc[[x for x in desc.columns if (x[1] != ('count')) | (x == ('PS_MW', 'count'))]]
    
def methods(df):   
    # cluster using pseudo f
    sweep_clustering(range(2,8), df, metric="Pseudo_f")

    # cluster using silhouette
    clusters = sweep_clustering(range(2,8), df, metric="Silhouette")

    # table of cluster stats     
    df["cluster_label"] = clusters
    assert hasattr(df,"cluster_label"),  len(clusters[0])
    describe_clusters(df) 

    # visualiztions of cluster distributions
    max_n = 5
    c = df.columns.tolist()
    x = 0
    while x < len(c) & max_n < len(c):
        cols = c[x:x+max_n]
        cols.append('cluster_label')
        cluster_dists(df[cols])
        x += max_n
        
    # describe clusters
    describe_clusters(df)
    return df.reset_index()
